```typescript
export async function confirmTransaction(id: string): Promise<Transaction> {
  const res = await fetch(`/api/transactions/${id}/confirm`, {
    method: 'POST',
  });
  if (!res.ok) throw new Error('Failed to confirm transaction');
  return res.json();
}
```
