<!-- From internal/handler/ (existing API endpoints) -->
Accounts:
```
GET    /api/accounts               → List all accounts
POST   /api/accounts               → Create account (body: {name, current_balance, currency})
GET    /api/accounts/:id           → Get account by ID
PATCH  /api/accounts/:id           → Update account (body: {name, current_balance, currency})
DELETE /api/accounts/:id           → Delete account
POST   /api/accounts/:id/set-default → Set as default account
```

Transactions:
```
GET    /api/transactions                    → List all transactions (optional: account_id query param)
POST   /api/transactions                    → Create transaction
                                              body: {account_id, amount, description, category, is_recurring, frequency, anchor_date}
PATCH  /api/transactions/:id                → Update transaction
DELETE /api/transactions/:id                → Delete transaction
```

Error responses:
```typescript
interface ErrorResponse {
  code: string      // e.g. "INVALID_BODY", "NOT_FOUND", "CONFLICT"
  error: string     // User-friendly message
}
```

// Accounts
fetchAccounts(): Promise<AccountResponse[]>

deleteAccount(id: string): Promise<void>
setDefaultAccount(id: string): Promise<void>

// Transactions
fetchTransactions(accountId?: string): Promise<TransactionResponse[]>
createTransaction(data: {account_id: string, amount: number, description: string, category: string, is_recurring?: boolean, frequency?: string, anchor_date?: string}): Promise<TransactionResponse>
updateTransaction(id: string, data: Partial<{amount, description, category, is_recurring, frequency, anchor_date}>): Promise<TransactionResponse>
deleteTransaction(id: string): Promise<void>

// Types
interface AccountResponse {

  current_balance: number
  currency: string
  is_default: boolean

  description: string
  category: string
  timestamp: string
  is_recurring: boolean

  next_occurrence: string | null
